121 static THD_FUNCTION(shell_thread, p) {
122  int n;
123  BaseSequentialStream *chp = ((ShellConfig *)p)->sc_channel;
124  const ShellCommand *scp = ((ShellConfig *)p)->sc_commands;
125  char *lp, *cmd, *tokp, line[SHELL_MAX_LINE_LENGTH];
126  char *args[SHELL_MAX_ARGUMENTS + 1];
127 
128  chRegSetThreadName("shell");
129  chprintf(chp, "\r\nChibiOS/RT Shell\r\n");
130  chprintf(chp, "Welcome !!!\r\n");
131  while (true) {
132 #if SHELL_PROMPT
133  chprintf(chp, "ch> ");
134 #endif
135  if (shellGetLine(chp, line, sizeof(line))) {
136  chprintf(chp, "\r\nlogout");
137  break;
138  }
139  lp = _strtok(line, " \t", &tokp);
140  cmd = lp;
141  n = 0;
142  while ((lp = _strtok(NULL, " \t", &tokp)) != NULL) {
143  if (n >= SHELL_MAX_ARGUMENTS) {
144  chprintf(chp, "too many arguments\r\n");
145  cmd = NULL;
146  break;
147  }
148  args[n++] = lp;
149  }
150  args[n] = NULL;
151  if (cmd != NULL) {
152  if (strcasecmp(cmd, "exit") == 0) {
153  if (n > 0) {
154  usage(chp, "exit");
155  continue;
156  }
157  break;
158  }
159  else if (strcasecmp(cmd, "help") == 0) {
160  if (n > 0) {
161  usage(chp, "help");
162  continue;
163  }
164  chprintf(chp, "Commands: help exit ");
165  list_commands(chp, local_commands);
166  if (scp != NULL)
167  list_commands(chp, scp);
168  chprintf(chp, "\r\n");
169  }
170  else if (cmdexec(local_commands, chp, cmd, n, args) &&
171  ((scp == NULL) || cmdexec(scp, chp, cmd, n, args))) {
172  chprintf(chp, "%s", cmd);
173  chprintf(chp, " ?\r\n");
174  }
175  }
176  }
177  shellExit(MSG_OK);
178 }

256 bool shellGetLine(BaseSequentialStream *chp, char *line, unsigned size) {
257  char *p = line;
258 
259  while (true) {
260  char c;
261 
262  if (chSequentialStreamRead(chp, (uint8_t *)&c, 1) == 0)
263  return true;
264  if (c == 4) {
265  chprintf(chp, "^D");
266  return true;
267  }
268  if ((c == 8) || (c == 127)) {
269  if (p != line) {
270  chSequentialStreamPut(chp, c);
271  chSequentialStreamPut(chp, 0x20);
272  chSequentialStreamPut(chp, c);
273  p--;
274  }
275  continue;
276  }
277  if (c == '\r') {
278 #if SHELL_ECHO
279  chprintf(chp, "\r\n");
280 #endif
281  *p = 0;
282  return false;
283  }
284  if (c < 0x20)
285  continue;
286  if (p < line + size - 1) {
287 #if SHELL_ECHO
288  chSequentialStreamPut(chp, c);
289 #endif
290  *p++ = (char)c;
291  }
292  }
293 }
