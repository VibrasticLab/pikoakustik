153 void ht_audio_Tone(double freq, double ampl){
154  uint16_t i;
155  uint16_t buffsize;
156  double ysin;
157 
158  buffsize = (uint16_t) I2S_BUFF_SIZE/freq;
159 
160  ht_audio_Zero();
161 
162  for(i=0;i<buffsize;i++){
163  ysin = DEFAULT_ATTEN*ampl*32767*sin(2*3.141592653589793*((double)i/(double)buffsize));
164  if(ysin >= 0){
165  i2s_tx_buf[i]=ysin;
166  }
167  if(ysin <0 ){
168  i2s_tx_buf[i]=ysin+65535;
169  }
170  }
171 
172  i2scfg.size = buffsize;
173 }

218 void ht_audio_Sine(double freq, double ampl, uint8_t model){
219  uint16_t i;
220  uint16_t halfsize;
221 
222  halfsize = (uint16_t) (buffsize/2)-1;
223 
224  ht_audio_Zero();
225 
226  if(model==0){
227  for(i=0;i<I2S_BUFF_SIZE;i++){
228  i2s_tx_buf[i] = ampl*DEFAULT_AMPLI*sin((double) freq*i*2*(M_PI/I2S_BUFF_SIZE));
229  }
230  }
231  else if(model==1){
232  //half upper/lower
233  i2s_tx_buf[i] = DEFAULT_ATTEN*ampl*32767*sin(3.141592653589793*((double)i/(double)halfsize));
234  i2s_tx_buf[halfsize+i] = 32767*(2-DEFAULT_ATTEN*ampl*sin(3.141592653589793*((double)i/(double)halfsize)));
235  }
236  else if(model==2){
237  // half lower
238  i2s_tx_buf[i] = DEFAULT_ATTEN*ampl*32767*sin(3.141592653589793*((double)i/(double)halfsize));
239  i2s_tx_buf[halfsize+i] = 65535;
240  }
241  else if(model==3){
242  // Half Up
243  i2s_tx_buf[i] = 0;
244  i2s_tx_buf[halfsize+i] = 32767*(2-DEFAULT_ATTEN*ampl*sin(3.141592653589793*((double)i/(double)halfsize)));
245  }
